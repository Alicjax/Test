    # Setup class and configuration
    def setUp(self):
        desired_caps = {
            "platformName": "Android",
            "deviceName": "emulator-5554",
            "appPackage": "com.inditex.zara",
            "appActivity": "com.inditex.zara.MainActivity"
        }
        # Start the Appium client
        self.driver = webdriver.Remote("http://localhost:4723/wd/hub", desired_caps)
        self.driver.implicitly_wait(20)

    # Close the Appium client
    def tearDown(self):
        self.driver.quit()

    def test1(self):
        # Enter on main page
        zara = self.driver.find_element(AppiumBy.ID, "com.inditex.zara:id/zaraButtonLabel")
        zara.click()
        # Click 'Continue'
        zaraContinue = self.driver.find_element(AppiumBy.XPATH, "/hierarchy/android.widget.FrameLayout/android.widget.LinearLayout/android.widget.FrameLayout/android.widget.LinearLayout/android.widget.FrameLayout/android.widget.RelativeLayout/android.widget.LinearLayout/android.widget.RelativeLayout/android.widget.FrameLayout/android.widget.FrameLayout/android.view.ViewGroup/android.widget.FrameLayout/android.widget.FrameLayout/android.view.ViewGroup/android.widget.LinearLayout/android.widget.LinearLayout")
        zaraContinue.click()
        # Click 'Don't allow"
        Donotallow = self.driver.find_element(AppiumBy.ID,"com.android.permissioncontroller:id/permission_deny_button")
        Donotallow.click()
        # 1. Kliknij "“ACCOUNT”
        account = self.driver.find_element(AppiumBy.ID,"com.inditex.zara:id/accountTab2")
        account.click()
        # 2. Kliknij “REGISTER”
        register = self.driver.find_element(AppiumBy.XPATH,"/hierarchy/android.widget.FrameLayout/android.widget.LinearLayout/android.widget.FrameLayout/android.widget.LinearLayout/android.widget.FrameLayout/android.widget.RelativeLayout/android.widget.LinearLayout/android.widget.FrameLayout/android.widget.RelativeLayout/android.widget.FrameLayout/android.view.ViewGroup/android.widget.ScrollView/android.view.ViewGroup/android.view.ViewGroup/android.widget.LinearLayout/android.widget.LinearLayout")
        register.click()
        # 3. Wpisz email
        email_field = self.driver.find_element(AppiumBy.ID,"com.inditex.zara:id/registerMailField")
        email_field.send_keys(self.email)
        # 4. Wpisz hasło
        hasło = self.driver.find_element(AppiumBy.ID,"com.inditex.zara:id/registerPasswordField")
        hasło.send_keys(self.password)
        # 5. Wpisz ponownie hasło
        hasło2 = self.driver.find_element(AppiumBy.ID, "com.inditex.zara:id/registerRepeatPasswordField")
        hasło2.send_keys(self.password)
        # 6. Wpisz imię używając cyfr
        name = self.driver.find_element(AppiumBy.XPATH,"/hierarchy/android.widget.FrameLayout/android.widget.LinearLayout/android.widget.FrameLayout/android.widget.LinearLayout/android.widget.FrameLayout/android.widget.RelativeLayout/android.widget.LinearLayout/android.widget.FrameLayout/android.widget.RelativeLayout/android.widget.FrameLayout/android.view.ViewGroup/android.widget.ScrollView/android.view.ViewGroup/android.widget.LinearLayout[1]/android.widget.LinearLayout[1]/android.widget.AutoCompleteTextView[1]")
        name.send_keys(self.invalid_name)
        # 7. Wpisz nazwisko używając cyfr
        lastname = self.driver.find_element(AppiumBy.XPATH, "/hierarchy/android.widget.FrameLayout/android.widget.LinearLayout/android.widget.FrameLayout/android.widget.LinearLayout/android.widget.FrameLayout/android.widget.RelativeLayout/android.widget.LinearLayout/android.widget.FrameLayout/android.widget.RelativeLayout/android.widget.FrameLayout/android.view.ViewGroup/android.widget.ScrollView/android.view.ViewGroup/android.widget.LinearLayout[1]/android.widget.LinearLayout[1]/android.widget.AutoCompleteTextView[2]")
        lastname.send_keys(self.invalid_surname)

        acceptterms = self.driver.find_element(AppiumBy.ID,"com.inditex.zara:id/addressItemCheckbox")
        acceptterms.click()
        # 9. Kliknij “CREATE ACCOUNT”
        createaccount = self.driver.find_element(AppiumBy.XPATH,"/hierarchy/android.widget.FrameLayout/android.widget.LinearLayout/android.widget.FrameLayout/android.widget.LinearLayout/android.widget.FrameLayout/android.widget.RelativeLayout/android.widget.LinearLayout/android.widget.FrameLayout/android.widget.RelativeLayout/android.widget.FrameLayout/android.view.ViewGroup/android.widget.ScrollView/android.view.ViewGroup/android.widget.LinearLayout[2]/android.widget.LinearLayout")
